    <div class="card no-print">

        <h3>Wheel Settings</h3>

        <div id="segmentsList" class="controls">

<script>

    let categories = ["Career", "Finance", "Health", "Friends", "Family", "Romance", "Growth", "Fun"];

    let nowScores = [5, 4, 7, 6, 8, 3, 5, 6];

    let wantScores = [8, 8, 9, 8, 9, 8, 9, 9];

    let chart;


    function initChart() {

        const ctx = document.getElementById('wheelChart').getContext('2d');

        chart = new Chart(ctx, {

            type: 'radar',

            data: {

                labels: categories,

                datasets: [{

                    label: 'Where I am NOW',

                    data: nowScores,

                    fill: true,

                    backgroundColor: 'rgba(37, 99, 235, 0.2)',

                    borderColor: 'rgb(37, 99, 235)',

                    pointBackgroundColor: 'rgb(37, 99, 235)',

                }, {

                    label: 'Where I WANT to be',

                    data: wantScores,

                    fill: true,

                    backgroundColor: 'rgba(16, 185, 129, 0.1)',

                    borderColor: 'rgb(16, 185, 129)',

                    borderDash: [5, 5],

                    pointBackgroundColor: 'rgb(16, 185, 129)',

                }]

            },

            options: {

                scales: { r: { min: 0, max: 10, ticks: { stepSize: 1 } } },

                plugins: { legend: { position: 'bottom' } }

            }

        });

    }


    function renderControls() {

        const list = document.getElementById('segmentsList');

        list.innerHTML = '';

        categories.forEach((cat, i) => {

            const row = document.createElement('div');

            row.className = 'segment-row';

            row.innerHTML = `

                <input type="text" value="${cat}" oninput="updateLabel(${i}, this.value)">

                <div><label>Now</label><input type="range" min="0" max="10" value="${nowScores[i]}" oninput="updateScore(${i}, 'now', this.value)"></div>

                <div><label>Goal</label><input type="range" min="0" max="10" value="${wantScores[i]}" oninput="updateScore(${i}, 'want', this.value)"></div>

                <button class="btn-del" onclick="removeSegment(${i})">×</button>

            `;

            list.appendChild(row);

        });

    }


    function updateLabel(index, val) {

        categories[index] = val;

        chart.data.labels = categories;

        chart.update();

    }


    function updateScore(index, type, val) {

        if(type === 'now') nowScores[index] = val;

        else wantScores[index] = val;

        chart.data.datasets[0].data = nowScores;

        chart.data.datasets[1].data = wantScores;

        chart.update();

    }


    function addSegment() {

        categories.push("New Item");

        nowScores.push(5);

        wantScores.push(8);

        renderControls();

        chart.update();

    }


    function removeSegment(index) {

        categories.splice(index, 1);

        nowScores.splice(index, 1);

        wantScores.splice(index, 1);

        renderControls();

        chart.update();

    }


    initChart();

    renderControls();

</script>
